<script>
// ── State ──────────────────────────────────────────────────────────────────
let currentType = 'MARKET';
let currentSide = 'BUY';
let connected   = false;

// ── Type selector ──────────────────────────────────────────────────────────
document.querySelectorAll('.type-item').forEach(el => {
  el.addEventListener('click', () => {
    document.querySelectorAll('.type-item').forEach(x => x.classList.remove('active'));
    el.classList.add('active');
    currentType = el.dataset.type;
    renderDynamicFields();
    updateSubmitBtn();
    document.getElementById('form-title').textContent = typeLabel(currentType);
  });
});

function typeLabel(t) {
  const map = {
    MARKET:'Market Order', LIMIT:'Limit Order', STOP_MARKET:'Stop-Market Order',
    STOP_LIMIT:'Stop-Limit Order', OCO:'OCO Order', TWAP:'TWAP Order', GRID:'Grid Order'
  };
  return map[t] || t;
}

// ── Side toggle ────────────────────────────────────────────────────────────
function setSide(side) {
  currentSide = side;
  document.getElementById('btn-buy').classList.toggle('active',  side === 'BUY');
  document.getElementById('btn-sell').classList.toggle('active', side === 'SELL');
  updateSubmitBtn();
}

function updateSubmitBtn() {
  const btn = document.getElementById('submit-btn');
  const label = currentSide === 'BUY' ? 'Buy' : 'Sell';
  btn.textContent = `${label} · ${typeLabel(currentType)}`;
  btn.className = `submit-btn ${currentSide === 'BUY' ? 'buy' : 'sell'}`;
}

// ── Dynamic fields ─────────────────────────────────────────────────────────
const fieldDefs = {
  price:           { label: 'Limit Price',        type: 'number', placeholder: 'e.g. 95000',  id: 'price' },
  stop_price:      { label: 'Stop Price',          type: 'number', placeholder: 'e.g. 94000',  id: 'stop_price' },
  stop_limit_price:{ label: 'Stop Limit Price',    type: 'number', placeholder: 'e.g. 93800',  id: 'stop_limit_price' },
  time_in_force:   { label: 'Time in Force',       type: 'select', options: ['GTC','IOC','FOK'], id: 'time_in_force' },
  slices:          { label: 'Slices',              type: 'number', placeholder: '5',            id: 'slices', min: 2, step: 1 },
  interval:        { label: 'Interval (seconds)',  type: 'number', placeholder: '10',           id: 'interval', min: 1, step: 1 },
  child_type:      { label: 'Child Order Type',    type: 'select', options: ['MARKET','LIMIT'], id: 'child_type' },
  levels:          { label: 'Grid Levels',         type: 'number', placeholder: '5',            id: 'levels', min: 2, step: 1 },
  start_price:     { label: 'Start Price',         type: 'number', placeholder: 'e.g. 95000',  id: 'start_price' },
  step:            { label: 'Step (price gap)',     type: 'number', placeholder: 'e.g. 500',    id: 'step' },
};

const typeFields = {
  MARKET:      [],
  LIMIT:       ['price', 'time_in_force'],
  STOP_MARKET: ['stop_price'],
  STOP_LIMIT:  ['stop_price', 'price', 'time_in_force'],
  OCO:         ['price', 'stop_price', 'stop_limit_price'],
  TWAP:        ['slices', 'interval', 'child_type'],
  GRID:        ['levels', 'start_price', 'step'],
};

function renderDynamicFields() {
  const container = document.getElementById('dynamic-fields');
  const fields = typeFields[currentType] || [];
  container.innerHTML = fields.map(f => {
    const d = fieldDefs[f];
    if (d.type === 'select') {
      const opts = d.options.map(o => `<option>${o}</option>`).join('');
      return `<div class="field"><label>${d.label}</label><select id="${d.id}">${opts}</select></div>`;
    }
    const extras = d.min  !== undefined ? `min="${d.min}"` : '';
    const step   = d.step !== undefined ? `step="${d.step}"` : 'step="any"';
    return `<div class="field"><label>${d.label}</label><input type="${d.type}" id="${d.id}" placeholder="${d.placeholder}" ${extras} ${step}></div>`;
  }).join('');
}

// ── Credentials ────────────────────────────────────────────────────────────
document.getElementById('settings-toggle').addEventListener('click', () => {
  document.getElementById('settings-section').classList.toggle('open');
});

async function checkSavedCredentials() {
  // On page load, check if credentials are already saved/in env
  try {
    const res  = await fetch('/api/credentials');
    const data = await res.json();
    if (data.status === 'ok' && data.data.configured) {
      // Try connecting silently
      const pingRes  = await fetch('/api/credentials', {
        method: 'POST',
        headers: {'Content-Type':'application/json'},
        body: JSON.stringify({api_key: '', api_secret: '', save: false}),
      });
      const pingData = await pingRes.json();
      if (pingData.status === 'ok') {
        markConnected(pingData.data.source, pingData.data.server_time);
        showCredBanner(`Auto-connected from: ${data.data.source}`);
        return;
      }
    }
  } catch(e) {}
  // Not connected yet — open settings
  document.getElementById('settings-section').classList.add('open');
}

function markConnected(source, serverTime) {
  connected = true;
  document.getElementById('conn-dot').classList.add('live');
  document.getElementById('conn-text').textContent = 'Connected · Testnet';
  document.getElementById('settings-section').classList.remove('open');
  addLog('ok', 'Connected', {source: source || 'configured', server_time: serverTime || '—'});
}

function showCredBanner(msg) {
  const banner = document.getElementById('cred-source-banner');
  banner.textContent = '✔ ' + msg;
  banner.style.display = 'block';
}

async function saveCredentials() {
  const key    = document.getElementById('api-key').value.trim();
  const secret = document.getElementById('api-secret').value.trim();
  const save   = document.getElementById('save-creds').checked;

  if (!key || !secret) { addLog('error', 'Credentials', {error: 'Both API key and secret are required'}); return; }

  addLog('info', 'Connecting…', {api_key: key.slice(0,8)+'…', save_to_disk: save});
  const res  = await fetch('/api/credentials', {
    method: 'POST',
    headers: {'Content-Type':'application/json'},
    body: JSON.stringify({api_key: key, api_secret: secret, save}),
  });
  const data = await res.json();
  if (data.status === 'ok') {
    const savedMsg = save ? ' · saved to disk' : ' · session only';
    markConnected(data.data.source, data.data.server_time);
    showCredBanner((save ? 'Credentials saved — will auto-load next time' : 'Connected for this session only'));
    addLog('ok', 'Connected' + savedMsg, {server_time: data.data.server_time});
  } else {
    addLog('error', 'Connection failed', {error: data.message});
  }
}

async function clearCredentials() {
  if (!confirm('Delete saved credentials from disk?')) return;
  const res  = await fetch('/api/credentials', {method: 'DELETE'});
  const data = await res.json();
  connected  = false;
  document.getElementById('conn-dot').classList.remove('live');
  document.getElementById('conn-text').textContent = 'Not connected';
  document.getElementById('api-key').value    = '';
  document.getElementById('api-secret').value = '';
  document.getElementById('cred-source-banner').style.display = 'none';
  addLog('info', 'Credentials cleared', {disk: data.data?.cleared ? 'deleted' : 'none found'});
}

// ── Submit ─────────────────────────────────────────────────────────────────
async function submitOrder() {
  if (!connected) {
    addLog('error', 'Not connected', {hint: 'Open API Credentials and click Connect first'});
    return;
  }

  const btn = document.getElementById('submit-btn');
  btn.disabled = true;
  btn.innerHTML = '<span class="spinner"></span>Placing order…';

  const payload = {
    symbol:     document.getElementById('symbol').value.trim().toUpperCase(),
    side:       currentSide,
    order_type: currentType,
    quantity:   document.getElementById('quantity').value,
  };

  // Collect dynamic fields
  (typeFields[currentType] || []).forEach(f => {
    const el = document.getElementById(f);
    if (el) payload[f] = el.value;
  });

  addLog('info', `${currentType} · ${currentSide}`, {
    symbol: payload.symbol, qty: payload.quantity, type: currentType,
  });

  const res  = await fetch('/api/order', {
    method: 'POST',
    headers: {'Content-Type':'application/json'},
    body: JSON.stringify(payload),
  });
  const data = await res.json();

  btn.disabled = false;
  updateSubmitBtn();

  if (data.status === 'ok') {
    const r = Array.isArray(data.data) ? data.data[0] : data.data;
    const rows = {};
    if (r) {
      if (r.orderId)    rows.orderId    = r.orderId;
      if (r.status)     rows.status     = r.status;
      if (r.executedQty && r.executedQty !== '0') rows.executedQty = r.executedQty;
      if (r.avgPrice && r.avgPrice !== '0' && r.avgPrice !== '0.00000000') rows.avgPrice = r.avgPrice;
      if (r.price && r.price !== '0' && r.price !== '0.00000000') rows.price = r.price;
    }
    if (Array.isArray(data.data)) rows['slices/levels'] = data.data.length;
    addLog('ok', `${currentType} placed`, rows);
  } else {
    addLog('error', `${currentType} failed`, {error: data.message});
  }
}

// ── Log panel ──────────────────────────────────────────────────────────────
function addLog(tag, title, rows) {
  const list = document.getElementById('log-list');
  const now  = new Date().toLocaleTimeString();
  const rowsHtml = Object.entries(rows).map(([k, v]) => {
    const cls = (k === 'status' && v === 'FILLED') ? 'green'
              : (k === 'error') ? 'red' : '';
    return `<div class="log-row"><span class="log-key">${k}</span><span class="log-val ${cls}">${v}</span></div>`;
  }).join('');

  const entry = document.createElement('div');
  entry.className = 'log-entry';
  entry.innerHTML = `
    <div class="log-header">
      <span class="log-tag ${tag}">${tag.toUpperCase()}</span>
      <span class="log-time">${now}</span>
    </div>
    <div class="log-body">${rowsHtml}</div>
  `;
  list.prepend(entry);

  // Keep last 30 entries
  while (list.children.length > 30) list.removeChild(list.lastChild);
}
